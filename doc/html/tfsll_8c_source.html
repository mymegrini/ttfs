285 fileblock_add(disk_id id, uint32_t vol_addr, uint32_t inode, _index index)
286 {
287  error e;
288  uint32_t b_addr;
289  tfs_ftent ftent;
290  block b;
291 
292  //case 1: we need to add direct block
293  if (index->direct < TFS_DIRECT_BLOCKS_NUMBER-1){
294  //obtain freeblock
295  if((e = freeblock_pop(id, vol_addr, &b_addr))!= EXIT_SUCCESS) return e;
296  //update <index>
297  index->direct++;
298  //read and update <ftent>
299  if ((e = read_ftent(id, vol_addr, inode, &ftent))!=EXIT_SUCCESS) return e;
300  ftent.tfs_direct[index->direct]= b_addr;
301  //write <ftent> to disk
302  if ((e = write_ftent(id, vol_addr, inode, &ftent))!=EXIT_SUCCESS) return e;
303  return EXIT_SUCCESS;
304  }
305 
306  //case 2: we need to add first indirect1 block
307  if (index->direct == TFS_DIRECT_BLOCKS_NUMBER-1){
308  //obtain freeblock for indirect1 block
309  if((e = freeblock_pop(id, vol_addr, &b_addr))!= EXIT_SUCCESS) return e;
310  //update <index>
311  index->direct++;
312  index->indirect1_addr = b_addr;
313  index->indirect1 = -1; //empty indirect1 block
314  //read and update <ftent>
315  if ((e = read_ftent(id, vol_addr, inode, &ftent))!=EXIT_SUCCESS) return e;
316  ftent.tfs_indirect1= b_addr;
317  //write <ftent> to disk
318  if ((e = write_ftent(id, vol_addr, inode, &ftent))!=EXIT_SUCCESS) return e;
319  //relaunch function to add data block to indirect1 block (case 5)
320  return fileblock_add(id, vol_addr, inode, index);
321  }
322 
323  //case 3: we need to add indirect2 block
324  if (index->indirect1 == INT_PER_BLOCK-1 && index->indirect2 == -1){
325  //obtain freeblock for indirect2
326  if((e = freeblock_pop(id, vol_addr, &b_addr))!= EXIT_SUCCESS) return e;
327  //read and update <ftent>
328  if ((e = read_ftent(id, vol_addr, inode, &ftent))!=EXIT_SUCCESS) return e;
329  ftent.tfs_indirect2= b_addr;
330  //write <ftent> to disk
331  if ((e = write_ftent(id, vol_addr, inode, &ftent))!=EXIT_SUCCESS) return e;
332  //relaunch function to add first indirect1 block to indirect2 block (case 4)
333  index->indirect1++; //guarantees that we land in case 4
334  return fileblock_add(id, vol_addr, inode, index);
335  }
336 
337  //case 4: we need to a add indirect1 block to indirect2 block
338  if (index->indirect1 == INT_PER_BLOCK){
339  //obtain freeblock for indirect1
340  if((e = freeblock_pop(id, vol_addr, &b_addr))!= EXIT_SUCCESS) return e;
341  //update <index>
342  index->indirect1_addr = b_addr;
343  index->indirect2++;
344  index->indirect1 = -1; //empty indirect1 block
345  //update indirect2 block
346  b = new_block();
347  if ((e = read_block(id, b, index->indirect2_addr))!=EXIT_SUCCESS) return e;
348  wintle(index->indirect1_addr, b, index->indirect2);
349  if ((e = write_block(id, b, index->indirect2_addr))!=EXIT_SUCCESS) return e;
350  free(b);
351  //relaunch function to add first data block to new indirect1 block (case 5)
352  return fileblock_add(id, vol_addr, inode, index);
353  }
354 
355  //case 5: we need to add a data block to current indirect1 block
356  if (index->indirect1 < INT_PER_BLOCK-1){
357  //obtain free block
358  if((e = freeblock_pop(id, vol_addr, &b_addr))!= EXIT_SUCCESS) return e;
359  //update <index>
360  index->indirect1++;
361  //update indirect1 block
362  b = new_block();
363  if ((e = read_block(id, b, index->indirect1_addr))!=EXIT_SUCCESS) return e;
364  wintle(b_addr, b, index->indirect1);
365  if ((e = write_block(id, b, index->indirect1_addr))!=EXIT_SUCCESS) return e;
366  free(b);
367  return EXIT_SUCCESS;
368  }
369 
370  //case 6: corrupted index
371  return I_CORRUPTED;
372 }

375 fileblock_rm(disk_id id, uint32_t vol_addr, uint32_t inode, _index index)
376 {
377  error e;
378  tfs_ftent ftent;
379  block b;
380  uint32_t b_addr;
381 
382  //case 0: file is empty
383  if(index->direct == -1) return F_EMPTY;
384 
385  //case 1: we need to remove a direct block
386  if (index->direct < TFS_DIRECT_BLOCKS_NUMBER){
387  //find direct block address
388  if ((e = read_ftent(id, vol_addr, inode, &ftent))!=EXIT_SUCCESS) return e;
389  b_addr = ftent.tfs_direct[index->direct];
390  //free direct block
391  if((e = freeblock_push(id, vol_addr, b_addr))!= EXIT_SUCCESS) return e;
392  //update <index>
393  index->direct--;
394  return EXIT_SUCCESS;
395  }
396 
397  //case 2: we need to remove data block from indirect1 block
398  if (index->indirect1 > -1){
399  //read data block address from indirect1
400  b = new_block();
401  if ((e = read_block(id, b, index->indirect1_addr))!=EXIT_SUCCESS) return e;
402  rintle(&b_addr, b, index->indirect1);
403  //free data block
404  if((e = freeblock_push(id, vol_addr, index->indirect2))!= EXIT_SUCCESS) return e;
405  //update <index>
406  index->indirect1--;
407  free(b);
408  //relaunch function if current indirect1 empty (cases 3 and 5)
409  if (index->indirect1== -1) return fileblock_rm(id, vol_addr, inode, index);
410  else return EXIT_SUCCESS;
411  }
412 
413  //case 3: we need to remove indirect1 block from indirect2 block
414  if (index->indirect1 == -1 && index->indirect2 > 0){
415  //free for indirect1
416  if((e = freeblock_push(id, vol_addr, index->indirect1_addr))!= EXIT_SUCCESS)
417  return e;
418  //update index
419  index->indirect2--;
420  index->indirect1+= INT_PER_BLOCK;
421  //find new indirect1 address from indirect2
422  b = new_block();
423  if ((e = read_block(id, b, index->indirect2_addr))!=EXIT_SUCCESS) return e;
424  rintle(&(index->indirect1_addr), b, index->indirect2);
425  free(b);
426  return EXIT_SUCCESS;
427  }
428 
429  //case 4: we need to remove last indirect1 block from indirect2 block
430  if (index->indirect1 == -1 && index->indirect2 == 0){
431  //free blocks indirect1 and indirect2
432  if((e = freeblock_push(id, vol_addr, index->indirect1_addr))!= EXIT_SUCCESS)
433  return e;
434  if((e = freeblock_push(id, vol_addr, index->indirect2_addr))!= EXIT_SUCCESS)
435  return e;
436  //update <index>
437  index->indirect2--;
438  index->indirect1+= INT_PER_BLOCK;
439  //find new indirect1 address from <ftent>
440  if ((e = read_ftent(id, vol_addr, inode, &ftent))!=EXIT_SUCCESS) return e;
441  index->indirect1 = ftent.tfs_indirect1;
442  return EXIT_SUCCESS;
443  }
444 
445  //case 5: we need to remove last indirect1 block
446  if (index->indirect1 == -1 && index->direct==TFS_DIRECT_BLOCKS_NUMBER){
447  //obtain free block
448  if((e = freeblock_pop(id, vol_addr, &b_addr))!= EXIT_SUCCESS) return e;
449  //update <index>
450  index->direct--;
451  return EXIT_SUCCESS;
452  }
453 
454  //case 6: corrupted index
455  return I_CORRUPTED;
456 }

460  tfs_description tfs_d;
461  error e;
462 
463  //get semaphore
464  char name[SEM_NAME_LEN];
465  sem_name(name, SEM_FBL_T, id, vol_addr, 0);
466  sem_t* sem = sem_open(name, O_CREAT,
467  S_IRUSR|S_IWUSR|S_IRGRP|S_IWGRP|S_IROTH|S_IWOTH, 1);
468  sem_wait(sem);
469 
470  //read volume superblock
471  if ((e = read_tfsdescription(id, vol_addr, &tfs_d))!=EXIT_SUCCESS) return e;
472  else {
473  block b= new_block();
474 
475  //fill appropriate TFS_VOLUME_NEXT_FREE_BLOCK_INDEX value in <b>
476  if (tfs_d.freeb_count == 0)
477  wintle(b_addr, b, TFS_VOLUME_NEXT_FREE_BLOCK_INDEX);
478  else
479  wintle(tfs_d.freeb_first, b, TFS_VOLUME_NEXT_FREE_BLOCK_INDEX);
480 
481  //write <b> in <b_addr> volume block
482  if ((e = write_block(id, b, vol_addr + b_addr))!=EXIT_SUCCESS){
483  free(b);
484  //free semaphore
485  sem_post(sem);
486  sem_close(sem);
487  return e;
488  }
489  free(b);
490 
491  //update <tfs_d> and write it to disk
492  tfs_d.freeb_first = b_addr;
493  tfs_d.freeb_count++;
494  e= write_tfsdescription(id, vol_addr, &tfs_d);
495 
496  //free semaphore
497  sem_post(sem);
498  sem_close(sem);
499 
500  return e;
501  }

505 freeblock_pop (disk_id id, uint32_t vol_addr, uint32_t* b_addr) {
506  tfs_description desc;
507  error e;
508 
509  //get semaphore
510  char name[SEM_NAME_LEN];
511  sem_name(name, SEM_FBL_T, id, vol_addr, 0);
512  sem_t* sem = sem_open(name, O_CREAT,
513  S_IRUSR|S_IWUSR|S_IRGRP|S_IWGRP|S_IROTH|S_IWOTH, 1);
514  sem_wait(sem);
515 
516  //read volume superblock
517  if ((e = read_tfsdescription(id, vol_addr, &desc))!= EXIT_SUCCESS) return e;
518 
519  //test if volume is full
520  if (desc.freeb_count == 0) return V_FULL;
521  else {
522  uint32_t nextfreeb;
523  block b = new_block();
524 
525  //read first free volume block
526  e = read_block(id, b, vol_addr + desc.freeb_first);
527  if (e != EXIT_SUCCESS) {
528  free(b);
529  //free semaphore
530  sem_post(sem);
531  sem_close(sem);
532  return e;
533  }
534 
535  //store block address in <b_addr>
536  *b_addr = desc.freeb_first;
537 
538  //determine next free block index
539  rintle(&nextfreeb, b, TFS_VOLUME_NEXT_FREE_BLOCK_INDEX);
540  free(b);
541 
542  //test if we have reached the end of the list
543  if (nextfreeb == *b_addr && desc.freeb_count >1){
544  //free semaphore
545  sem_post(sem);
546  sem_close(sem);
547  return V_FBL_CORRUPTED;
548  } else {
549  //update tfs_description structure
550  desc.freeb_first = nextfreeb;
551  desc.freeb_count--;
552 
553  //free semaphore
554  sem_post(sem);
555  sem_close(sem);
556 
557  //update superblock on disk
558  return write_tfsdescription(id, vol_addr, &desc);
559  }
560  }
